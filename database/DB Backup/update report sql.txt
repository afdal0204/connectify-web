✅ Cara cek data mana yang menyebabkan error

Jalankan query ini:

SELECT model_id
FROM reports
LEFT JOIN models ON reports.model_id = models.id
WHERE reports.model_id IS NOT NULL
  AND models.id IS NULL;

SELECT model_id
FROM reports
LEFT JOIN devices ON reports.device_id = devices.id
WHERE reports.device_id IS NOT NULL
  AND devices.id IS NULL;

SELECT r.*
FROM reports r
LEFT JOIN error_code e ON r.error_code_id = e.id
WHERE r.error_code_id IS NOT NULL
  AND e.id IS NULL;
